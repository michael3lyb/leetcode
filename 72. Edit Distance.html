public class Solution {
    public int minDistance(String word1, String word2) {
        int m = word1.length();
        int n = word2.length();

        int[][] f = new int[m + 1][n + 1];
        for (int i = 0;i <= m; i++) {
            f[i][0] = i;
        }
        for (int i = 0;i <= n; i++) {
            f[0][i] = i;
        }
        int temp = 0;
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (word1.charAt(i - 1) == word2.charAt(j - 1)) {
                    //这里不需要都比较
                    // temp = Math.min(f[i - 1][j] + 1, f[i][j - 1] + 1);
                    // f[i][j] = Math.min(f[i - 1][j - 1], temp);
                    f[i][j] = f[i - 1][j - 1];
                } else {
                    temp = Math.min(f[i - 1][j] + 1, f[i][j - 1] + 1); // delete and insertion
                    f[i][j] = Math.min(f[i - 1][j - 1] + 1, temp);  // mismatch
                }
            }
        }
        return f[m][n];
    }
}

public class Solution {
    public int minDistance(String word1, String word2) {
        int m = word1.length();
        int n = word2.length();

        int[][] cost = new int[m + 1][n + 1];
        for(int i = 0; i <= m; i++)
            cost[i][0] = i;
        for(int i = 1; i <= n; i++)
            cost[0][i] = i;

        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                if(word1.charAt(i) == word2.charAt(j))
                    cost[i + 1][j + 1] = cost[i][j];
                else {
                    int a = cost[i][j];
                    int b = cost[i][j + 1];
                    int c = cost[i + 1][j];
                    cost[i + 1][j + 1] = a < b ? (a < c ? a : c) : (b < c ? b : c);
                    cost[i + 1][j + 1]++;
                }
            }
        }
        return cost[m][n];
    }
}
